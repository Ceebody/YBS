  let studentName = "";
  let currentQuestion = 0;
  let answers = [];
  let timer;
  let timeLeft = 15 * 60; 
  const lockDuration = 12 * 60 * 60 * 1000;

  const namePage = document.getElementById("namePage");
  const welcomePage = document.getElementById("welcomePage");
  const countdownScreen = document.getElementById("countdownScreen");
  const quizArea = document.getElementById("quizArea");
  const resultsDiv = document.getElementById("results");
  const questionDiv = document.getElementById("question");
  const optionsDiv = document.getElementById("options");
  const shortAnswer = document.getElementById("shortAnswer");
  const prevBtn = document.getElementById("prevBtn");
  const nextBtn = document.getElementById("nextBtn");
  const submitBtn = document.getElementById("submitBtn");
  const timerDiv = document.getElementById("timer");
  const lockMsg = document.getElementById("lockMessage");

  document.getElementById("proceedBtn").addEventListener("click", async () => {
    const nameInput = document.getElementById("studentName").value.trim();
    if (nameInput === "") {
      alert("Please enter your name before proceeding!");
      return;
    }
    studentName = nameInput;

    const userRef = ref(db, "quizResults/" + studentName);
    const snapshot = await get(userRef);
    if (snapshot.exists()) {
      const data = snapshot.val();
      if (data.lockedUntil && Date.now() < data.lockedUntil) {
        const unlockTime = new Date(data.lockedUntil).toLocaleString();
        lockMsg.textContent = `You already completed this quiz. You can try again after ${unlockTime}`;
        return;
      }
    }

    namePage.classList.add("hidden");
    welcomePage.classList.remove("hidden");
  });

  document.getElementById("startBtn").addEventListener("click", () => {
    welcomePage.classList.add("hidden");
    startCountdown();
  });

  function startCountdown() {
    countdownScreen.classList.remove("hidden");
    let counter = 3;
    countdownScreen.textContent = "Get ready... " + counter;

    const countdown = setInterval(() => {
      counter--;
      if (counter > 0) {
        countdownScreen.textContent = counter + "...";
      } else {
        clearInterval(countdown);
        countdownScreen.textContent = "Begin!";
        setTimeout(() => {
          countdownScreen.classList.add("hidden");
          startQuiz();
        }, 800);
      }
    }, 1000);
  }

  function startQuiz() {
    quizArea.classList.remove("hidden");
    loadQuestion();
    startTimer();
  }

  function loadQuestion() {
    const q = quizData[currentQuestion];
    questionDiv.innerHTML = `<h2>${q.q}</h2>`;
    optionsDiv.innerHTML = "";
    shortAnswer.classList.add("hidden");

    if (q.options) {
      q.options.forEach(opt => {
        const btn = document.createElement("button");
        btn.textContent = opt;
        btn.className = "secondary";
        btn.onclick = () => saveAnswer(opt);
        if (answers[currentQuestion] === opt) {
          btn.style.background = "#4CAF50";
          btn.style.color = "white";
        }
        optionsDiv.appendChild(btn);
      });
    }

    prevBtn.disabled = currentQuestion === 0;
    nextBtn.classList.toggle("hidden", currentQuestion === quizData.length - 1);
    submitBtn.classList.toggle("hidden", currentQuestion !== quizData.length - 1);
  }

  function saveAnswer(ans) {
    answers[currentQuestion] = ans;
    loadQuestion();
  }

  nextBtn.onclick = () => {
    if (currentQuestion < quizData.length - 1) {
      currentQuestion++;
      loadQuestion();
    }
  };

  prevBtn.onclick = () => {
    if (currentQuestion > 0) {
      currentQuestion--;
      loadQuestion();
    }
  };

  submitBtn.onclick = finishQuiz;

  function startTimer() {
    timer = setInterval(() => {
      timeLeft--;
      let mins = Math.floor(timeLeft / 60);
      let secs = timeLeft % 60;
      timerDiv.textContent = `Time Left: ${mins}:${secs.toString().padStart(2, "0")}`;
      if (timeLeft <= 0) {
        clearInterval(timer);
        finishQuiz();
      }
    }, 1000);
  }

  function finishQuiz() {
    clearInterval(timer);
    quizArea.classList.add("hidden");
    resultsDiv.classList.remove("hidden");

    let score = 0;
    let summaryHTML = "<div class='summary'><h3>Summary</h3>";

    quizData.forEach((q, i) => {
      const userAns = answers[i] || "(no answer)";
      const correct = userAns === q.answer;
      if (correct) score++;

     summaryHTML += `<p><strong>Q${i + 1}:</strong> ${q.q}<br>
    Your answer: <span class="${correct ? 'correct' : 'wrong'}">${userAns}</span><br>
    Correct answer: <span class="correct">${q.answer}</span><br>
    <span style="color:#333;"><em>Work:</em> ${q.work}</span></p>`;


    const percentage = ((score / quizData.length) * 100).toFixed(2);
    summaryHTML += "</div>";

    resultsDiv.innerHTML = `
      <h2>Quiz Completed!</h2>
      <p><strong>Name:</strong> ${studentName}</p>
      <p>Your Score: <strong>${score} / ${quizData.length}</strong> (${percentage}%)</p>
      ${summaryHTML}
    `;

    const lockedUntil = Date.now() + lockDuration;
    const quizRef = ref(db, "quizResults/" + studentName);
    set(quizRef, {
      name: studentName,
      score: score,
      percentage: percentage,
      totalQuestions: quizData.length,
      timestamp: new Date().toISOString(),
      lockedUntil: lockedUntil
    });
  }
